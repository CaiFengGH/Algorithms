>题目描述

将十进制数转换为二进制数，统计二进制数字中包含1的个数

- 实现方法(增加十进制数转换为二进制数)

```
/**
 * @author Ethan
 * @desc 数值转换的技巧
 */
public class NumOfOne {
	/**
	 * @desc 统计十进制数的二进制格式中1的数量
	 * @param num 十进制数
	 * @return 1的数量
	 */
	public static int numOfOne(int num){
		//初始化1的数量
		int count = 0;
		while(num != 0){
			count++;
			//采用相与的方式
			num = (num - 1) & num;
		}
		return count;
	}
	
	/**
	 * @desc 除2取余实现十进制到二进制的转变 
	 * @param num 十进制数
	 */
	public static void decimalToBinary(int num){
		//初始化存储数组，数组元素，除2余数
		int[] arr = new int[32];
		//反向打印数组时，不从数组末尾开始打印
		int numItems = 0;
		int remain = 0;
		//循环取余数直至为0
		while(num > 0){
			if( num % 2 == 1){
				remain = 1;
			}else{
				remain = 0;
			}
			arr[numItems++] = remain;
			num = num / 2;
		}
		//反向打印出存储余数的数组
		for(int j = numItems - 1; j >= 0; j--){
			System.out.println(arr[j]);
		}
	}
}
```
